import heapq

class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        
        if not intervals:
            return 0
        
        def fun(interval):
            return interval[0]
        
        intervals.sort(key = fun)
        
        prev, count = 0, 0
        
        for i in range(1, len(intervals)):
            if intervals[i][0] >= intervals[prev][1]:
                prev = i
            else:
                if intervals[i][1] > intervals[prev][1]:
                    prev = prev
                else:
                    prev = i
                count +=1
                
        return count
