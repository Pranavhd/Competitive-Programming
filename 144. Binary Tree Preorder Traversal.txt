# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def preorderTraversal(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        
        ret = []
        stack = [root]
        
        while stack:
            curr = stack.pop()
            if curr:
                ret.append(curr.val)
                stack.append(curr.right)
                stack.append(curr.left)
        return ret
